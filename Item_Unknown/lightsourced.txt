[TRIG 3026]
//Interior Candle Sconce (north wall)
//by Thadius - Tranquillus Decorum

ONTRIGGER=DCLICK
MOREP=<SRC.P>
MOREX=<MOREX>
MOREY=<MOREY> - 1
MOREZ=<MOREZ> + 13
SRC.HOME=<MOREP>
SRC.NEWITEM=0a00
//Candle Sconce N
SRC.ACT.ATTR=8010
SRC.ACT.P=<SRC.HOME>
SRC.ACT.MOREZ=2
//That sets the light type and radius
SOUND 0042

REMOVE
RETURN 1
